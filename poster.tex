  \headerbox{tl;dr}{name=tldr,column=0}{
    \vskip .1in
    {\raggedright\normalsize
    \BI
    \item CloudCoder is a free, open-source,
          web-based programming exercise system
    \item It supports C/C++, Java, Python, and Ruby
    \item It has a repository of free exercises
          (contributions welcome)
    \item Ask for an account on our demo server
    \item Did we mention that it's free?
    \item \url{http://cloudcoder.org}
    \EI
    }
  }

  \headerbox{Programming exercises!}{name=exercises,column=0,below=tldr}{
    \vskip .1in
    {\raggedright\normalsize
    \BI
    \item Automatically-evaluated exercises are useful in introductory
          programming courses
      \BI
      \item For learning syntax
      \item For learning basic concepts
            (variables, decisions, loops, etc.)
      \item For extra practice
      \item For assessment
      \EI
    \item They can be incorporated easily into an existing
          course without major changes
      \BI
      \item Use them to supplement readings, for quizzes,
            for practice, etc.
      \EI
    \EI
    }
  }

  \headerbox{We need another system?}{name=anothersystem,column=0,below=exercises}{
    \vskip .1in
    {\raggedright\normalsize
    \BI
    \item Lots of existing systems:
          CodingBat, PracticeIt!, CodeLab, etc.
    \item We didn't know of one which
      \BI
      \item Was open source
      \item Was community-supported
      \item Had freely-redistributable exercises
      \item Allowed self-hosting and full access to
            student data
      \item Supported in-class assessment (i.e., quizzes)
      \item Supported C/C++
      \EI
    \item So we wrote one
    \EI
    }
  }

  \headerbox{Need for a community}{name=community,column=1}{
    \vskip .1in
    {\raggedright\normalsize
    \BI
    \item Thesis: for a programming exercise system
          to be as useful as it can be, it needs a {\it community}
      \BI
      \item To provide exercises that can be freely
            reused and improved
      \item To share experience about how to use
            exercises most effectively
      \EI
    \EI
    }
